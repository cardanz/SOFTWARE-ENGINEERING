181 def switch_fun(self,argument): 

182 """ Funzione di switch per le varie funzioni possibili

184 Parametri

197 # il nome della stringa scelta dall'interfaccia

199 switcher = {

201 'sigmoid': tf.nn.sigmoid,

205 'tanh': tf.nn.tanh,

212 def generazioneModello(self): 

213 """ Generazione della Rete Neurale con le caratteristiche scelte """ 

214 

215 # salvataggio numero nodi e layers e della funzione di attivazione 

216 nNodi = self.reteNeurale.numeroNodi 

217 nLayer = self.reteNeurale.numeroLayer 

218 model = self.reteNeurale.model 

219 funzioneAtt = self.switch_fun(self.reteNeurale.funzione) 

220 

221 # crezione del modello di rete 

222 model = tf.keras.Sequential(name = self.reteNeurale.modelName) 

223 

224 # aggiunta del layer iniziale 

225 model.add(tf.keras.layers.Dense(nNodi, activation=funzioneAtt, input_shape=(4,))) 

226 

227 # aggiunta di 'i' ( numero scelto da interfaccia ) di hidden layer 

228 for i in range(1, nLayer + 1): 

229 model.add(tf.keras.layers.Dense(nNodi, activation=funzioneAtt)) 

230 

231 # aggiunta del output layer 

232 model.add(tf.keras.layers.Dense(3)) 

233 

234 # setto il modello creato all'oggetto reteNeurale 

235 self.reteNeurale.model = model

237 def trainingModello(self): 

238 """ Training della Rete Neurale """ 

239 # creo la funzione di loss (scelgo la SparseCategoricalCrossentropy per via della 

240 # presenza di 2 o piu label) 

241 loss_object = tf.keras.losses.SparseCategoricalCrossentropy(from_logits=True) 

242 

243 # definisco la funzione 'loss' che confronta la label  

244 # letta nel dataset con quella ottentuta dal training 

245 def loss(model, x, y, training): 

246 y_ = model(x, training=training) 

247 return loss_object(y_true=y, y_pred=y_) 

248 

249 def grad(model, inputs, targets): 

250 with tf.GradientTape() as tape: 

251 loss_value = loss(model, inputs, targets, training=True) 

252 

253 return loss_value, tape.gradient(loss_value, model.trainable_variables) 

254 

255 # scelgo l' ottimizzatore 

256 optimizer = tf.keras.optimizers.SGD(learning_rate=0.01) 

257 

258 # salvo i dati della rete che devo allenare 

259 model = self.reteNeurale.model 

260 features = self.reteNeurale.features 

261 labels = self.reteNeurale.labels 

262 

263 # setto il numero di 'epoch' (epoche) per cui 

264 # devo riperete il train 

265 num_epochs = 201 

266 

267 # training effettivo 

268 for epoch in range(num_epochs): 

269 # definisco la loss media e la accuratezza 

270 epoch_loss_avg = tf.keras.metrics.Mean() 

271 epoch_accuracy = tf.keras.metrics.SparseCategoricalAccuracy() 

272 

273 # training loop 

274 for x, y in self.reteNeurale.train_dataset: 

275 

276 # ottimizzo il modello , salvando la loss 

277 # e il gradiente 

278 loss_value, grads = grad(model, x, y) 

279 

280 optimizer.apply_gradients(zip(grads, model.trainable_variables)) 

281 

282 # tengo l' andamento della loss per creare il grafico 

283 epoch_loss_avg.update_state(loss_value) 

284 # e anche della accuracy 

285 epoch_accuracy.update_state(y, model(x, training=True)) 

286 

287 # salvataggio completo delle variabili  

288 # contenenti l andamento della loss e della accuracy 

289 self.reteNeurale.train_loss_results.append(epoch_loss_avg.result()) 

290 self.reteNeurale.train_accuracy_results.append(epoch_accuracy.result()) 

291 

292 if epoch % 50 == 0: 

293 print("Epoch {:03d}: Loss: {:.3f}, Accuracy: {:.3%}".format(epoch, 

294 epoch_loss_avg.result(), 

295 epoch_accuracy.result()))

336 """ Effettua la prediction sul file scelto

338 Parametri 

339 --------- 

340 predictionPath: string 

341 Percorso del file contenente il dataset su cui si deve 

342 effettuare la 'prediction' 

343 

344 

345 Ritorna 

346 ------- 

347 ris: string[] 

348 Risultati della 'prediction' nella forma: 

349 "Riga: " + i + 

350 "Previsione: " + nomeFiorePredetto + percentualediAccuratezza

353 # leggo il dataset di previsione 

354 pre_dataset = genfromtxt(predictionPath, delimiter=',')

360 ris = []

362 for i, logits in enumerate(predictions): 

363 class_idx = tf.argmax(logits).numpy() 

364 funzioneAtt = tf.nn.softmax 

365 p = funzioneAtt(logits)[class_idx] 

366 name = self.class_names[class_idx] 

367 

368 # aggiunta dei valori predetti, con la percentuale di accuracy 

369 ris.append("Riga: " + str(i) + "\nPrevisione: " + name +" ("+"{:.3%}".format(p) + ")") 

370 

371 return ris
